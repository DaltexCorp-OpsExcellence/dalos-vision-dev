function applyInspectionPrefill(selectedIds,allResults){ var selected=allResults.filter(function(r){return selectedIds.indexOf(r.id)>=0;}); if(selected.length===0) return; var phSeen={}, farmSeen={}, varSeen={}, tcSeen={}; var phList=[], farmList=[], varList=[], tcList=[]; selected.forEach(function(r){ /* Packhouse - store name for dropdown */ if(r.packhouse_id&&!phSeen[r.packhouse_id]){ phSeen[r.packhouse_id]=true; var phName=(config.packhouses&&config.packhouses.find(function(p){return p.id===r.packhouse_id;})&&config.packhouses.find(function(p){return p.id===r.packhouse_id;}).name)||r.packhouse_id; phList.push(phName); } /* Farms + plots */ if(r.farm_id&&!farmSeen[r.farm_id]){ farmSeen[r.farm_id]=true; var fn=(config.farmsIdToName&&config.farmsIdToName[r.farm_id])||r.farm_id; farmList.push({name:fn,plot:r.plot||''}); } if(r.farm2_id&&!farmSeen[r.farm2_id]){ farmSeen[r.farm2_id]=true; var fn2=(config.farmsIdToName&&config.farmsIdToName[r.farm2_id])||r.farm2_id; farmList.push({name:fn2,plot:r.plot2||''}); } /* Varieties */ if(r.variety_id){ var vn=(config.varietiesIdToName&&config.varietiesIdToName[r.variety_id])||''; var vn2=r.variety2_id?(config.varietiesIdToName&&config.varietiesIdToName[r.variety2_id])||'':''; var vkey=r.variety_id+(r.variety2_id||''); if(!varSeen[vkey]){varSeen[vkey]=true;varList.push({var:vn,var2:vn2,type:r.variety_type||'single'});} } /* Trace codes - collect unique from trace_code (text) and trace_codes (jsonb array) */ var allTc=[]; if(r.trace_code) allTc.push(r.trace_code); if(r.trace_codes&&Array.isArray(r.trace_codes)) r.trace_codes.forEach(function(tc){ if(tc&&typeof tc==='string') allTc.push(tc); else if(tc&&tc.code) allTc.push(tc.code); }); allTc.forEach(function(tc){if(tc&&!tcSeen[tc]){tcSeen[tc]=true;tcList.push(tc);}}); /* Wax - first inspection wins */ if(r.wax&&!fWax) setWax(r.wax); /* GGN */ if(r.ggn&&!fTrace) setTrace(r.ggn); /* Country/destination - first inspection wins */ if(r.country_id&&!fRegion){ var ctName=(config.countriesIdToName&&config.countriesIdToName[r.country_id])||''; if(ctName) setRegion(ctName); } /* Daltex class - first inspection wins */ if(r.class_id&&!fDaltexCls){ var clsName=(config.classesIdToName&&config.classesIdToName[r.class_id])||''; if(clsName) setDaltexCls(clsName); } }); /* Apply packhouses - fPH stores name for dropdown */ if(phList.length>0){ setPH(phList[0]); setPhCover(phList[0]); setPHExtra(phList.slice(1).map(function(n){return {actual:n,cover:''};})); } /* Apply farms */ if(farmList.length>0){ setFarm1(farmList[0].name);setPlot1(farmList[0].plot||''); if(farmList.length>1){setFarm2(farmList[1].name);setPlot2(farmList[1].plot||'');} if(farmList.length>2) setFarmsExtra(farmList.slice(2).map(function(f){return {farm:f.name,plot:f.plot||''};})); } /* Apply trace codes - each one as a separate entry */ if(tcList.length>0) setTraceCodes(tcList.map(function(c){return {code:c};})); /* Apply varieties */ if(varList.length>0){ var v0=varList[0];setVar(v0.var);setVarType(v0.type||'single');setVar2(v0.var2||''); var allVars=(config.varietiesFull&&config.varietiesFull[selCqcProd||'grapes'])||[]; var found=allVars.find(function(v){return v.name===v0.var;}); if(found&&found.categoryId){var cn=(config.categoriesIdToName||{})[found.categoryId]||'';if(cn)setFCat(cn);} if(varList.length>1){ setVarsExtra(varList.slice(1).map(function(ve){ var catName=''; var fnd=allVars.find(function(v){return v.name===ve.var;}); if(fnd&&fnd.categoryId)catName=(config.categoriesIdToName||{})[fnd.categoryId]||''; var cat2Name=''; if(ve.var2){var fnd2=allVars.find(function(v){return v.name===ve.var2;});if(fnd2&&fnd2.categoryId)cat2Name=(config.categoriesIdToName||{})[fnd2.categoryId]||'';} return {type:ve.type||'single',cat:catName,var:ve.var,cat2:cat2Name,var2:ve.var2||''}; })); } } /* Pack date from first inspection */ if(selected[0].date&&!fPackDate) setPackDate(selected[0].date); }
